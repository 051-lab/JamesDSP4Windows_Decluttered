cmake_minimum_required(VERSION 3.16)
project(JamesDSP-Windows VERSION 1.0.0 LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_C_STANDARD 11)

# Windows-specific settings
if(WIN32)
    add_definitions(-D_CRT_SECURE_NO_WARNINGS)
    add_definitions(-DWIN32_LEAN_AND_MEAN)
    add_definitions(-DNOMINMAX)
endif()

# libjamesdsp sources
set(LIBJDSP_DIR "${CMAKE_SOURCE_DIR}/../JDSP4Linux/libjamesdsp/subtree/Main/libjamesdsp/jni/jamesdsp")

file(GLOB_RECURSE LIBJDSP_SOURCES 
    "${LIBJDSP_DIR}/jdsp/*.c"
    "${LIBJDSP_DIR}/cpthread.c"
)

# Create static library for libjamesdsp
add_library(jamesdsp STATIC ${LIBJDSP_SOURCES})

target_include_directories(jamesdsp PUBLIC
    "${CMAKE_SOURCE_DIR}/src/compat"
    "${CMAKE_SOURCE_DIR}/../JDSP4Linux/libjamesdsp"
    "${LIBJDSP_DIR}"
    "${LIBJDSP_DIR}/jdsp"
    "${LIBJDSP_DIR}/jdsp/Effects"
    "${LIBJDSP_DIR}/jdsp/Effects/eel2"
    "${LIBJDSP_DIR}/jdsp/generalDSP"
)

target_compile_definitions(jamesdsp PRIVATE
    _USE_MATH_DEFINES
    WIN32
    _WIN32
)

# Main executable
add_executable(JamesDSP-Console
    src/main.cpp
    src/DspConfig.cpp
)

target_link_libraries(JamesDSP-Console PRIVATE
    jamesdsp
    ole32
    uuid
)

target_include_directories(JamesDSP-Console PRIVATE
    "${LIBJDSP_DIR}"
    "${LIBJDSP_DIR}/jdsp"
)
